# Task: 数据验证模块开发

## Description
开发全面的输入数据质量检查和清洗模块，确保Shapefile数据的几何有效性、属性完整性和坐标系统一致性。实现自动化数据质量评估、错误检测和修复建议，为后续的空间关联处理提供高质量的数据基础。

## Acceptance Criteria
- [ ] 实现几何数据有效性检查（自相交、无效几何、空几何等）
- [ ] 开发属性数据完整性验证（缺失值、数据类型、值域范围等）
- [ ] 构建坐标系统一致性检查和自动修复功能
- [ ] 实现数据质量评分和报告生成系统
- [ ] 开发自动化数据清洗和修复工具
- [ ] 支持自定义验证规则和阈值配置

```python
class GeometryValidator:
    """几何数据有效性检查器"""

    def __init__(self, tolerance=1e-6):
        self.tolerance = tolerance
        self.validation_rules = self._initialize_validation_rules()
        self.error_counts = defaultdict(int)
        self.error_details = []

    def _initialize_validation_rules(self):
        """初始化验证规则"""
        return {
            'valid_geometry': self._check_valid_geometry,
            'no_self_intersection': self._check_no_self_intersection,
            'proper_orientation': self._check_proper_orientation,
            'minimum_area': self._check_minimum_area,
            'coordinate_bounds': self._check_coordinate_bounds,
            'no_duplicates': self._check_no_duplicates
        }

    def validate_geometries(self, gdf, geometry_type=None):
        """验证几何数据"""
        validation_result = GeometryValidationResult()
        validation_result.total_count = len(gdf)

        print(f"开始验证 {len(gdf)} 个几何对象...")

        for idx, row in gdf.iterrows():
            geometry = row.geometry
            geom_errors = []

            # 跳过空几何
            if geometry is None or geometry.is_empty:
                validation_result.add_error(idx, 'empty_geometry', '几何对象为空')
                continue

            # 应用所有验证规则
            for rule_name, rule_func in self.validation_rules.items():
                if not self._should_apply_rule(rule_name, geometry_type):
                    continue

                try:
                    is_valid, error_message = rule_func(geometry, row)
                    if not is_valid:
                        geom_errors.append((rule_name, error_message))
                        self.error_counts[rule_name] += 1

                except Exception as e:
                    geom_errors.append(('rule_error', f'规则 {rule_name} 执行失败: {str(e)}'))

            if geom_errors:
                validation_result.add_geometry_errors(idx, geom_errors)
            else:
                validation_result.valid_count += 1

        return validation_result

    def _check_valid_geometry(self, geometry, row):
        """检查几何对象有效性"""
        if not geometry.is_valid:
            try:
                # 尝试修复几何对象
                fixed_geometry = geometry.buffer(0)
                if fixed_geometry.is_valid:
                    return False, f"几何对象无效但可修复: {geometry.geom_type}"
                else:
                    return False, f"几何对象无法修复: {geometry.geom_type}, 错误: {ExplainValid(geometry)}"
            except:
                return False, f"几何对象严重无效: {geometry.geom_type}"

        return True, None

    def _check_no_self_intersection(self, geometry, row):
        """检查自相交"""
        if hasattr(geometry, 'intersects') and hasattr(geometry, 'buffer'):
            # 检查几何对象是否自相交
            if geometry.geom_type in ['LineString', 'Polygon']:
                # 创建一个微小缓冲区来检测自相交
                buffer_geom = geometry.buffer(self.tolerance)
                if buffer_geom.area < geometry.area * 1.1:  # 面积变化过小可能表示自相交
                    return False, "检测到自相交"

        return True, None

    def _check_proper_orientation(self, geometry, row):
        """检查几何方向（多边形外环应该是顺时针）"""
        if geometry.geom_type == 'Polygon':
            exterior = geometry.exterior
            if not self._is_clockwise(exterior.coords):
                return False, "多边形外环方向错误（应为顺时针）"

        elif geometry.geom_type == 'MultiPolygon':
            for poly in geometry.geoms:
                exterior = poly.exterior
                if not self._is_clockwise(exterior.coords):
                    return False, "多边形集合中有外环方向错误"

        return True, None

    def _is_clockwise(self, coords):
        """判断坐标序列是否为顺时针方向"""
        if len(coords) < 4:  # 至少需要4个点（包含重复的起点）
            return True

        # 使用鞋带公式计算有向面积
        area = 0
        for i in range(len(coords) - 1):
            area += (coords[i+1][0] - coords[i][0]) * (coords[i+1][1] + coords[i][1])

        return area > 0  # 面积为正表示顺时针

    def _check_minimum_area(self, geometry, row):
        """检查最小面积要求"""
        if hasattr(geometry, 'area'):
            if geometry.area < self.tolerance:
                return False, f"几何对象面积过小: {geometry.area}"

        return True, None

    def _check_coordinate_bounds(self, geometry, row):
        """检查坐标范围"""
        bounds = geometry.bounds
        min_x, min_y, max_x, max_y = bounds

        # 检查坐标是否在合理范围内（中国境内）
        if not (70 < min_x < 140 and 0 < min_y < 60):
            return False, f"坐标范围异常: {bounds}"

        # 检查坐标是否有效（非NaN、非无穷）
        if any(coord != coord or abs(coord) == float('inf') for bound in bounds for coord in [bound]):
            return False, "包含无效坐标值（NaN或Inf）"

        return True, None

    def _check_no_duplicates(self, geometry, row):
```

```python
class GeometryRepairer:
    """几何对象修复工具"""

    def __init__(self, tolerance=1e-6):
        self.tolerance = tolerance
        self.repair_strategies = {
            'invalid_geometry': self._repair_invalid_geometry,
            'self_intersection': self._repair_self_intersection,
            'wrong_orientation': self._repair_orientation,
            'tiny_geometry': self._repair_tiny_geometry
        }

    def repair_geometry(self, geometry, error_types):
        """修复几何对象"""
        repaired_geometry = geometry
        repair_log = []

        for error_type in error_types:
            if error_type in self.repair_strategies:
                try:
                    new_geometry, repair_message = self.repair_strategies[error_type](repaired_geometry)
                    if new_geometry is not None and new_geometry.is_valid:
                        repaired_geometry = new_geometry
                        repair_log.append(f"修复 {error_type}: {repair_message}")
                    else:
                        repair_log.append(f"修复 {error_type} 失败")
                except Exception as e:
                    repair_log.append(f"修复 {error_type} 时出错: {str(e)}")

        return repaired_geometry, repair_log

    def _repair_invalid_geometry(self, geometry):
        """修复无效几何对象"""
        try:
            # 方法1：使用buffer(0)修复
            fixed_geometry = geometry.buffer(0)
            if fixed_geometry.is_valid:
                return fixed_geometry, "使用buffer(0)修复"

            # 方法2：简化几何对象
            simplified = geometry.simplify(self.tolerance)
            if simplified.is_valid:
                return simplified, "简化几何对象修复"

            # 方法3：重新构建多边形
            if geometry.geom_type == 'Polygon':
                coords = list(geometry.exterior.coords)
                if len(coords) >= 4:
                    # 移除重复点
                    unique_coords = []
                    for coord in coords:
                        if not unique_coords or coord != unique_coords[-1]:
                            unique_coords.append(coord)

                    # 确保闭合
                    if unique_coords[0] != unique_coords[-1]:
                        unique_coords.append(unique_coords[0])

                    rebuilt = Polygon(unique_coords)
                    if rebuilt.is_valid:
                        return rebuilt, "重新构建多边形修复"

        except Exception:
            pass

        return None, "无法修复无效几何对象"

    def _repair_self_intersection(self, geometry):
        """修复自相交几何对象"""
        try:
            if geometry.geom_type == 'LineString':
                # 对于线，可以尝试简化
                simplified = geometry.simplify(self.tolerance)
                if not simplified.crosses(simplified):
                    return simplified, "简化线对象修复自相交"

            elif geometry.geom_type == 'Polygon':
                # 对于多边形，使用buffer(0)通常能修复自相交
                fixed = geometry.buffer(0)
                if fixed.is_valid and not self._has_self_intersection(fixed):
                    return fixed, "使用buffer(0)修复多边形自相交"

        except Exception:
            pass

        return None, "无法修复自相交"

    def _has_self_intersection(self, geometry):
        """检查几何对象是否自相交"""
        try:
            # 创建一个微小的缓冲区，如果面积增加过快可能表示自相交
            buffer_geom = geometry.buffer(self.tolerance)
            original_area = geometry.area if hasattr(geometry, 'area') else 0
            buffer_area = buffer_geom.area if hasattr(buffer_geom, 'area') else 0

            return buffer_area > original_area * 1.5  # 经验阈值
        except:
            return False
```

```python
class AttributeValidator:
    """属性数据完整性检查器"""

    def __init__(self):
        self.validation_config = self._load_validation_config()
        self.type_converters = self._initialize_type_converters()

    def validate_attributes(self, gdf, layer_name):
        """验证属性数据"""
        validation_result = AttributeValidationResult()

        # 检查必要字段
        self._check_required_fields(gdf, layer_name, validation_result)

        # 检查数据类型
        self._check_data_types(gdf, layer_name, validation_result)

        # 检查值域范围
        self._check_value_ranges(gdf, layer_name, validation_result)

        # 检查唯一性约束
        self._check_uniqueness_constraints(gdf, layer_name, validation_result)

        # 检查空值
        self._check_null_values(gdf, layer_name, validation_result)

        return validation_result

    def _check_required_fields(self, gdf, layer_name, validation_result):
        """检查必要字段"""
        required_fields = self.validation_config.get(layer_name, {}).get('required_fields', [])

        for field in required_fields:
            if field not in gdf.columns:
                validation_result.add_error('missing_field', f"缺少必要字段: {field}")
            elif gdf[field].isna().all():
                validation_result.add_error('empty_field', f"必要字段 {field} 全部为空")

    def _check_data_types(self, gdf, layer_name, validation_result):
        """检查数据类型"""
        expected_types = self.validation_config.get(layer_name, {}).get('field_types', {})

        for field, expected_type in expected_types.items():
            if field not in gdf.columns:
                continue

            actual_type = str(gdf[field].dtype)

            # 类型转换检查
            if not self._is_compatible_type(actual_type, expected_type):
                # 尝试自动转换
                try:
                    converted_series = self._convert_field_type(gdf[field], expected_type)
                    validation_result.add_suggestion(
                        'type_conversion',
                        f"字段 {field} 可以从 {actual_type} 转换为 {expected_type}"
                    )
                except Exception as e:
                    validation_result.add_error(
                        'type_mismatch',
                        f"字段 {field} 类型错误: 期望 {expected_type}, 实际 {actual_type}, 转换失败: {str(e)}"
                    )

    def _check_value_ranges(self, gdf, layer_name, validation_result):
        """检查值域范围"""
        range_constraints = self.validation_config.get(layer_name, {}).get('value_ranges', {})

        for field, constraints in range_constraints.items():
            if field not in gdf.columns:
                continue

            series = gdf[field].dropna()

            if 'min' in constraints:
                invalid_values = series[series < constraints['min']]
                if not invalid_values.empty:
                    validation_result.add_error(
                        'value_out_of_range',
                        f"字段 {field} 有 {len(invalid_values)} 个值小于最小值 {constraints['min']}"
                    )

            if 'max' in constraints:
                invalid_values = series[series > constraints['max']]
                if not invalid_values.empty:
                    validation_result.add_error(
                        'value_out_of_range',
                        f"字段 {field} 有 {len(invalid_values)} 个值大于最大值 {constraints['max']}"
                    )

    def _load_validation_config(self):
        """加载验证配置"""
        # 这里应该从配置文件加载，这里提供默认配置
        default_config = {
            'cross_section_points': {
                'required_fields': ['hecd', 'pcode', 'lgtd', 'lttd'],
                'field_types': {
                    'lgtd': 'float64',
                    'lttd': 'float64',
                    'ele': 'float64',
                    'cdistance': 'float64'
                },
                'value_ranges': {
                    'lgtd': {'min': 70, 'max': 140},
                    'lttd': {'min': 0, 'max': 60}
                }
            },
            'cross_section_lines': {
                'required_fields': ['名称', '编号', '河流名'],
                'field_types': {
                    '名称': 'string',
                    '编号': 'string',
                    '河流名': 'string'
                }
            }
        }

        return default_config
```

```python
class CoordinateSystemValidator:
    """坐标系统验证器"""

    def __init__(self):
        self.known_crs = {
            'WGS84': 'EPSG:4326',
            'CGCS2000': 'EPSG:4490',
            'WebMercator': 'EPSG:3857'
        }

    def validate_coordinate_system(self, gdf, expected_crs=None):
        """验证坐标系统"""
        validation_result = CRSValidationResult()

        # 检查CRS定义
        actual_crs = gdf.crs
        if actual_crs is None:
            validation_result.add_error('missing_crs', '数据集缺少坐标系统定义')
            return validation_result

        validation_result.actual_crs = str(actual_crs)

        # 如果指定了期望的CRS，检查一致性
        if expected_crs:
            if not self._is_same_crs(actual_crs, expected_crs):
                validation_result.add_error(
                    'crs_mismatch',
                    f'坐标系统不匹配: 期望 {expected_crs}, 实际 {actual_crs}'
                )
            else:
                validation_result.crs_consistent = True

        # 检查坐标范围是否符合CRS
        self._validate_coordinate_bounds(gdf, actual_crs, validation_result)

        return validation_result

    def _validate_coordinate_bounds(self, gdf, crs, validation_result):
        """根据CRS验证坐标范围"""
        bounds = gdf.total_bounds
        min_x, min_y, max_x, max_y = bounds

        # 根据不同的CRS检查坐标范围
        if self._is_geographic_crs(crs):
            # 地理坐标系（经纬度）
            if not (-180 <= min_x <= 180 and -180 <= max_x <= 180):
                validation_result.add_warning(

                    f'经度坐标超出有效范围: [{min_x}, {max_x}]'
                )

            if not (-90 <= min_y <= 90 and -90 <= max_y <= 90):
                validation_result.add_warning(
                    'latitude_out_of_range',
                    f'纬度坐标超出有效范围: [{min_y}, {max_y}]'
                )

        elif self._is_projected_crs(crs):
            # 投影坐标系
            # 这里需要根据具体的投影坐标系设置合理的范围
            pass

    def suggest_crs_transformation(self, gdf, target_crs):
        """建议坐标系统转换"""
        current_crs = gdf.crs

        if current_crs is None:
            return {
                'action': 'define_crs',
                'suggested_crs': self._suggest_crs_from_bounds(gdf),
                'confidence': 'medium'
            }

        if not self._is_same_crs(current_crs, target_crs):
            return {
                'action': 'transform',
                'source_crs': str(current_crs),
                'target_crs': target_crs,
                'transformation_needed': True
            }

        return {
            'action': 'no_action',
            'message': '当前坐标系统与目标一致'
        }

    def _suggest_crs_from_bounds(self, gdf):
        """根据坐标范围建议CRS"""
        bounds = gdf.total_bounds
        min_x, min_y, max_x, max_y = bounds

        # 如果坐标范围看起来像经纬度
        if (-180 <= min_x <= 180 and -90 <= min_y <= 90 and
            -180 <= max_x <= 180 and -90 <= max_y <= 90):

                # 中国境内

            else:
                return 'EPSG:4326'  # WGS84

        # 如果是大数值，可能是投影坐标系

            # 可能是投影坐标系，需要进一步判断
```

```python
class DataQualityScorer:
    """数据质量评分系统"""

    def __init__(self):
        self.scoring_weights = {
            'geometry_validity': 0.3,
            'attribute_completeness': 0.2,
            'coordinate_consistency': 0.2,
            'topological_correctness': 0.15,
            'metadata_quality': 0.15
        }

    def calculate_overall_score(self, validation_results):
        """计算总体质量评分"""
        scores = {}

        # 几何有效性评分
        scores['geometry_validity'] = self._score_geometry_validity(
            validation_results.get('geometry')
        )

        # 属性完整性评分
        scores['attribute_completeness'] = self._score_attribute_completeness(
            validation_results.get('attribute')
        )

        # 坐标一致性评分
        scores['coordinate_consistency'] = self._score_coordinate_consistency(
            validation_results.get('crs')
        )

        # 拓扑正确性评分
        scores['topological_correctness'] = self._score_topological_correctness(
            validation_results.get('topology')
        )

        # 元数据质量评分
        scores['metadata_quality'] = self._score_metadata_quality(
            validation_results.get('metadata')
        )

        # 计算加权总分
        overall_score = sum(
            scores[category] * self.scoring_weights[category]
            for category in self.scoring_weights
        )

        return {
            'overall_score': overall_score,
            'category_scores': scores,
            'grade': self._get_quality_grade(overall_score),
            'recommendations': self._generate_recommendations(scores)
        }

    def _score_geometry_validity(self, geometry_result):
        """几何有效性评分"""
        if not geometry_result:
            return 0.0

        total_features = geometry_result.total_count
        valid_features = geometry_result.valid_count

        if total_features == 0:
            return 0.0

        # 基础评分基于有效几何比例
        base_score = valid_features / total_features

        # 根据错误类型调整评分
        error_penalty = 0.0
        for error_type, count in geometry_result.error_counts.items():
            if error_type in ['invalid_geometry', 'self_intersection']:
                error_penalty += count * 0.02  # 严重错误扣分更多
            else:
                error_penalty += count * 0.01

        final_score = max(0, base_score - error_penalty)
        return min(1.0, final_score)

    def _get_quality_grade(self, score):
        """获取质量等级"""
        if score >= 0.95:
            return 'A'  # 优秀
        elif score >= 0.85:
            return 'B'  # 良好
        elif score >= 0.70:
            return 'C'  # 一般
        elif score >= 0.50:
            return 'D'  # 较差
        else:
            return 'F'  # 不合格

    def _generate_recommendations(self, scores):
        """生成改进建议"""
        recommendations = []

        for category, score in scores.items():
            if score < 0.7:
                if category == 'geometry_validity':
                    recommendations.append("建议运行几何修复工具处理无效几何对象")
                elif category == 'attribute_completeness':
                    recommendations.append("建议检查并补充缺失的属性数据")
                elif category == 'coordinate_consistency':
                    recommendations.append("建议统一坐标系统并进行必要的转换")
                elif category == 'topological_correctness':
                    recommendations.append("建议检查并修复拓扑错误")
                elif category == 'metadata_quality':
                    recommendations.append("建议完善数据集的元数据信息")

        return recommendations
```
