# Task: 结果输出和可视化模块

## Description

class ResultExporter:
    """结果数据导出框架"""

    def __init__(self):
        self.exporters = {
            'shapefile': ShapefileExporter(),
            'geojson': GeoJSONExporter(),
            'csv': CSVExporter(),
            'kml': KMLExporter(),
            'gpx': GPXExporter()
        }
        self.report_generator = ReportGenerator()

    def export_results(self, results, export_config):
        """导出处理结果"""
        exported_files = []
        export_dir = export_config.get('output_directory', './output')

        # 确保输出目录存在
        os.makedirs(export_dir, exist_ok=True)

        # 导出各个结果数据集
        for dataset_name, dataset_data in results.items():
            if not self._should_export_dataset(dataset_name, export_config):
                continue

            dataset_exports = self._export_dataset(
                dataset_name, dataset_data, export_config, export_dir
            )
            exported_files.extend(dataset_exports)

        # 生成统计报告
        if export_config.get('generate_report', True):
            report_file = self.report_generator.generate_report(results, export_dir)
            exported_files.append(report_file)

        return exported_files

    def _export_dataset(self, dataset_name, dataset_data, export_config, export_dir):
        """导出单个数据集"""
        exported_files = []
        formats = export_config.get('formats', ['shapefile'])

        for format_name in formats:
            if format_name not in self.exporters:
                print(f"警告: 不支持的导出格式 {format_name}")
                continue

            try:
                exporter = self.exporters[format_name]
                output_file = exporter.export(dataset_data, dataset_name, export_dir)
                exported_files.append(output_file)

                print(f"✅ 导出 {dataset_name} 到 {format_name} 格式: {output_file}")

            except Exception as e:
                print(f"❌ 导出 {dataset_name} 到 {format_name} 格式失败: {str(e)}")

        return exported_files

class ShapefileExporter:
    """Shapefile格式导出器"""

    def __init__(self):
        self.encoding = 'utf-8'

    def export(self, dataset_data, dataset_name, export_dir):
        """导出为Shapefile格式"""
        if not isinstance(dataset_data, (gpd.GeoDataFrame, pd.DataFrame)):
            raise ValueError("数据必须是GeoDataFrame或DataFrame")

        # 确保是GeoDataFrame
        if not isinstance(dataset_data, gpd.GeoDataFrame):
            raise ValueError("Shapefile导出需要几何数据")

        # 处理字段名长度限制（Shapefile字段名最多10个字符）
        gdf = self._prepare_shapefile_data(dataset_data)

        # 生成输出文件路径
        base_name = self._sanitize_filename(dataset_name)
        output_path = os.path.join(export_dir, f"{base_name}.shp")

        # 导出数据
        gdf.to_file(output_path, encoding=self.encoding, driver='ESRI Shapefile')

        # 创建.prj文件
        self._create_projection_file(gdf, output_path)

        return output_path

    def _prepare_shapefile_data(self, gdf):
        """准备Shapefile数据"""
        # 创建副本以避免修改原始数据
        prepared_gdf = gdf.copy()

        # 截断字段名
        column_mapping = {}
        for col in prepared_gdf.columns:
            if col != 'geometry' and len(col) > 10:
                new_name = col[:10]
                # 确保新名称唯一
                counter = 1
                while new_name in column_mapping.values():
                    new_name = col[:9] + str(counter)
                    counter += 1
                column_mapping[col] = new_name

        if column_mapping:
            prepared_gdf = prepared_gdf.rename(columns=column_mapping)

        # 处理字段类型（Shapefile支持的类型有限）
        prepared_gdf = self._convert_data_types(prepared_gdf)

        return prepared_gdf

    def _convert_data_types(self, gdf):
        """转换数据类型以符合Shapefile规范"""
        for col in gdf.columns:
            if col == 'geometry':
                continue

            # 将大整数转换为字符串
            if gdf[col].dtype == 'int64':
                gdf[col] = gdf[col].astype('int32')

            # 将浮点数转换为合理的精度
            elif gdf[col].dtype == 'float64':
                gdf[col] = gdf[col].astype('float32')

            # 处理字符串长度限制
            elif gdf[col].dtype == 'object':
                # Shapefile字符串字段最多254字符
                max_length = 254
                gdf[col] = gdf[col].astype(str).str.slice(0, max_length)

        return gdf

    def _create_projection_file(self, gdf, shp_path):
        """创建.prj投影文件"""
        if gdf.crs is not None:
            prj_path = shp_path.replace('.shp', '.prj')
            with open(prj_path, 'w') as f:
                f.write(gdf.crs.to_wkt())

class GeoJSONExporter:
    """GeoJSON格式导出器"""

    def export(self, dataset_data, dataset_name, export_dir):
        """导出为GeoJSON格式"""
        if not isinstance(dataset_data, gpd.GeoDataFrame):
            raise ValueError("GeoJSON导出需要几何数据")

        # 生成输出文件路径
        base_name = self._sanitize_filename(dataset_name)
        output_path = os.path.join(export_dir, f"{base_name}.geojson")

        # 确保使用WGS84坐标系（GeoJSON标准）
        gdf = dataset_data
        if gdf.crs != 'EPSG:4326':
            gdf = gdf.to_crs('EPSG:4326')

        # 导出数据
        gdf.to_file(output_path, driver='GeoJSON', encoding='utf-8')

        return output_path

class CSVExporter:
    """CSV格式导出器"""

    def export(self, dataset_data, dataset_name, export_dir):
        """导出为CSV格式"""
        # 生成输出文件路径
        base_name = self._sanitize_filename(dataset_name)
        output_path = os.path.join(export_dir, f"{base_name}.csv")

        # 处理数据
        if isinstance(dataset_data, gpd.GeoDataFrame):
            # 如果是GeoDataFrame，添加几何信息
            df = dataset_data.copy()

            # 添加几何类型
            df['geometry_type'] = df.geometry.geom_type

            # 添加坐标信息
            if df.geom_type.iloc[0] == 'Point':
                df['longitude'] = df.geometry.x
                df['latitude'] = df.geometry.y
            elif df.geom_type.iloc[0] in ['LineString', 'MultiLineString']:
                df['geometry_wkt'] = df.geometry.to_wkt()

            # 移除几何列
            df = df.drop('geometry', axis=1)
        else:
            df = dataset_data

        # 导出CSV
        df.to_csv(output_path, index=False, encoding='utf-8-sig')

        return output_path

        reports = {}

        html_report = self._generate_html_report(report_data, output_dir)
        reports['html'] = html_report

        md_report = self._generate_markdown_report(report_data, output_dir)
        reports['markdown'] = md_report

        # Excel统计报告
        excel_report = self._generate_excel_report(report_data, output_dir)
        reports['excel'] = excel_report

    def _collect_report_data(self, results):
        """收集报告数据"""

            'processing_time': time.time(),
            'datasets': {},
            'statistics': {},
            'quality_metrics': {},

                report_data['datasets'][dataset_name] = {
                    'type': 'geodataframe',
                    'geometry_type': dataset_data.geometry.geom_type.iloc[0] if len(dataset_data) > 0 else None,
                    'feature_count': len(dataset_data),
                    'columns': list(dataset_data.columns),
                    'bounds': dataset_data.total_bounds.tolist() if len(dataset_data) > 0 else None,
                    'crs': str(dataset_data.crs) if dataset_data.crs else None

                report_data['datasets'][dataset_name] = {
                    'type': 'dataframe',

        if 'associations' in results:
            report_data['associations'] = self._analyze_associations(results['associations'])

        # 计算质量指标
        report_data['quality_metrics'] = self._calculate_quality_metrics(results)

        return report_data

        return table_html

class DataVisualizer:
    """数据可视化模块"""

    def __init__(self):
        self.color_schemes = self._initialize_color_schemes()
        self.map_config = self._initialize_map_config()

    def create_visualization_dashboard(self, results, output_dir):

        dashboard_data = {
            'map_visualizations': [],
            'statistical_charts': [],
            'relationship_diagrams': []
        }

        # 创建地图可视化
        for dataset_name, dataset_data in results.items():
            if isinstance(dataset_data, gpd.GeoDataFrame):

                dashboard_data['map_visualizations'].append(map_viz)

        stats_charts = self._create_statistical_charts(results)

        # 创建关系图表
        if 'associations' in results:

            dashboard_data['relationship_diagrams'].append(relationship_diagram)

        # 生成HTML仪表板
        dashboard_path = self._generate_html_dashboard(dashboard_data, output_dir)
        return dashboard_path

    def _create_map_visualization(self, gdf, dataset_name):
        """创建地图可视化"""
        import folium
        import matplotlib.pyplot as plt
        import seaborn as sns

        # 确保使用WGS84坐标系用于地图显示
        if gdf.crs != 'EPSG:4326':
            gdf_wgs84 = gdf.to_crs('EPSG:4326')
        else:
            gdf_wgs84 = gdf

        # 计算地图中心点
        bounds = gdf_wgs84.total_bounds
        center_lat = (bounds[1] + bounds[3]) / 2
        center_lon = (bounds[0] + bounds[2]) / 2

        # 创建Folium地图
        m = folium.Map(
            location=[center_lat, center_lon],
            zoom_start=10,
            tiles='OpenStreetMap'
        )

        # 根据几何类型添加图层
        geom_type = gdf_wgs84.geometry.geom_type.iloc[0]

        if geom_type == 'Point':
            # 点图层
            self._add_point_layer(m, gdf_wgs84, dataset_name)
        elif geom_type == 'LineString':
            # 线图层
            self._add_line_layer(m, gdf_wgs84, dataset_name)
        elif geom_type == 'Polygon':
            # 面图层
            self._add_polygon_layer(m, gdf_wgs84, dataset_name)

        # 添加图层控制
        folium.LayerControl().add_to(m)

        # 保存地图
        map_path = f"output/{dataset_name}_map.html"
        m.save(map_path)

        return {
            'dataset_name': dataset_name,
            'map_path': map_path,
            'geometry_type': geom_type,
            'feature_count': len(gdf_wgs84)
        }

    def _add_point_layer(self, m, gdf, dataset_name):
        """添加点图层到地图"""
        # 生成颜色映射
        if len(gdf) > 1:
            # 使用数据值生成颜色
            numeric_columns = gdf.select_dtypes(include=['number']).columns
            if len(numeric_columns) > 0:
                color_column = numeric_columns[0]
                colors = plt.cm.viridis(gdf[color_column] / gdf[color_column].max())
            else:
                colors = ['blue'] * len(gdf)
        else:
            colors = ['blue']

        # 添加点到地图
        for idx, row in gdf.iterrows():
            folium.CircleMarker(
                location=[row.geometry.y, row.geometry.x],
                radius=5,
                popup=f"{dataset_name} #{idx}",
                color=colors[idx] if len(colors) > 1 else colors[0],
                fill=True,
                fillColor=colors[idx] if len(colors) > 1 else colors[0],
                fillOpacity=0.7
            ).add_to(m)

    def _add_line_layer(self, m, gdf, dataset_name):
        """添加线图层到地图"""
        for idx, row in gdf.iterrows():
            # 获取线坐标
            if hasattr(row.geometry, 'coords'):
                coords = list(row.geometry.coords)
                locations = [[coord[1], coord[0]] for coord in coords]  # 转换为[lat, lon]

                folium.PolyLine(
                    locations=locations,
                    popup=f"{dataset_name} #{idx}",
                    color='blue',
                    weight=2,
                    opacity=0.8
                ).add_to(m)

    def _add_polygon_layer(self, m, gdf, dataset_name):
        """添加面图层到地图"""
        for idx, row in gdf.iterrows():
            # 获取面坐标
            if hasattr(row.geometry, 'exterior'):
                coords = list(row.geometry.exterior.coords)
                locations = [[coord[1], coord[0]] for coord in coords]

                folium.Polygon(
                    locations=locations,
                    popup=f"{dataset_name} #{idx}",
                    color='blue',
                    fill=True,
                    fillColor='lightblue',
                    fillOpacity=0.5
                ).add_to(m)

    def _create_statistical_charts(self, results):
        """创建统计图表"""
        charts = []

        # 数据集大小分布图
        dataset_sizes = self._calculate_dataset_sizes(results)

        # 关关系统计图
        if 'associations' in results:

            charts.append(association_chart)

        # 数据质量评分图
        quality_chart = self._create_quality_chart(results)
        charts.append(quality_chart)

        return charts

    def _create_dataset_size_chart(self, dataset_sizes):
        """创建数据集大小分布图"""
        import matplotlib.pyplot as plt
        import seaborn as sns

        # 设置图表样式
        plt.style.use('seaborn-v0_8')
        fig, ax = plt.subplots(figsize=(10, 6))

        # 创建柱状图
        datasets = list(dataset_sizes.keys())
        sizes = list(dataset_sizes.values())

        bars = ax.bar(datasets, sizes, color='skyblue', alpha=0.7)

        ax.set_title('数据集要素数量分布', fontsize=16, fontweight='bold')
        ax.set_xlabel('数据集名称')
        ax.set_ylabel('要素数量')
        plt.xticks(rotation=45)
        plt.tight_layout()

        # 保存图表
        chart_path = 'output/dataset_size_distribution.png'

        plt.close()

        return {
            'chart_type': 'bar_chart',
            'title': '数据集要素数量分布',
            'path': chart_path
        }

    def _create_relationship_diagram(self, associations):
        """创建关联关系图"""
        import networkx as nx
        import matplotlib.pyplot as plt

        # 创建网络图
        G = nx.Graph()

        # 添加节点和边
        for assoc in associations:
            source = assoc.get('source', 'unknown')
            target = assoc.get('target', 'unknown')

            G.add_node(source)
            G.add_node(target)
            G.add_edge(source, target, weight=assoc.get('weight', 1))

        # 绘制关系图
        plt.figure(figsize=(12, 8))
        pos = nx.spring_layout(G, k=1, iterations=50)

        # 绘制节点
        nx.draw_networkx_nodes(G, pos, node_color='lightblue',
                             node_size=1000, alpha=0.7)

        # 绘制边
        nx.draw_networkx_edges(G, pos, alpha=0.5, width=2)

        # 绘制标签
        nx.draw_networkx_labels(G, pos, font_size=8, font_weight='bold')
