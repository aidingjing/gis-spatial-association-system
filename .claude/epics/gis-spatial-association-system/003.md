## Description

```python
class SpatialAssociationPipeline:
    """空间关联分析主管道"""

    def __init__(self, config):
        self.config = config
        self.stages = []
        self.state_manager = PipelineStateManager()
        self.error_handler = PipelineErrorHandler()
        self.progress_monitor = ProgressMonitor()
        self.data_validator = DataValidator()

        # 初始化处理阶段
        self._initialize_stages()

    def _initialize_stages(self):
        """初始化处理阶段"""
        self.stages = [
            DataLoadingStage(),
            DataPreprocessingStage(),
            CoordinateSystemUnificationStage(),
            PointLineAssociationStage(),
            LineLineIntersectionStage(),
            LinePolygonAssociationStage(),
            ResultPostProcessingStage(),
            DataExportStage()
        ]

    async def execute_pipeline(self, input_config):
        """执行完整的数据处理管道"""
        pipeline_start_time = time.time()

        try:
            # 管道初始化
            pipeline_id = self.state_manager.create_pipeline_session(input_config)
            self.progress_monitor.start_monitoring(pipeline_id)

            # 执行各个阶段
            context = PipelineContext(pipeline_id, input_config)

            for i, stage in enumerate(self.stages):
                stage_name = stage.__class__.__name__
                self.progress_monitor.update_stage_status(i, stage_name, 'starting')

                try:
                    # 阶段前置检查
                    if not await self._validate_prerequisites(stage, context):
                        raise PipelineError(f"Prerequisites not met for stage: {stage_name}")

                    # 执行阶段
                    stage_result = await stage.execute(context)

                    # 更新上下文
                    context.update_with_stage_result(stage_name, stage_result)

                    # 保存检查点
                    await self.state_manager.save_checkpoint(pipeline_id, i, context)

                    # 更新进度
                    self.progress_monitor.update_stage_status(i, stage_name, 'completed')

                except Exception as e:
                    # 阶段错误处理
                    error_info = self.error_handler.handle_stage_error(stage_name, e, context)
                    self.progress_monitor.update_stage_status(i, stage_name, 'failed', str(e))

                    if error_info['recoverable']:
                        # 尝试恢复
                        context = await self._attempt_recovery(error_info, context)
                    else:
                        # 不可恢复错误，终止管道
                        raise PipelineError(f"Unrecoverable error in stage {stage_name}: {str(e)}")

            # 管道完成
            pipeline_duration = time.time() - pipeline_start_time
            final_result = context.get_final_result()

            await self.state_manager.complete_pipeline(pipeline_id, final_result, pipeline_duration)
            self.progress_monitor.complete_monitoring(pipeline_id, True)

            return final_result

        except Exception as e:
            # 管道级别错误处理
            self.progress_monitor.complete_monitoring(pipeline_id, False, str(e))
            await self.state_manager.fail_pipeline(pipeline_id, str(e))
            raise
```

```python
class PointLineAssociationStage(BaseProcessingStage):
    """点-线关联处理阶段"""

    def __init__(self):
        super().__init__()
        self.associator = None
        self.performance_tracker = PerformanceTracker()

    async def execute(self, context: PipelineContext):
        """执行点-线关联"""
        self.performance_tracker.start_stage_timer()

        try:
            # 获取输入数据
            cross_section_points = context.get_data('cross_section_points')
            cross_section_lines = context.get_data('cross_section_lines')

            # 数据验证
            validation_result = await self.validate_input_data(cross_section_points, cross_section_lines)
            if not validation_result.is_valid:
                raise DataValidationError(f"Point-line association input validation failed: {validation_result.errors}")

            # 初始化关联器
            self.associator = NearestNeighborAssociator(
                use_spatial_index=context.config.get('use_spatial_index', True)
            )

            # 执行关联处理
            self.progress_monitor.update_progress(0, len(cross_section_points))

            associated_points = []
            chunk_size = context.config.get('chunk_size', 1000)

            for i in range(0, len(cross_section_points), chunk_size):
                chunk_end = min(i + chunk_size, len(cross_section_points))
                point_chunk = cross_section_points.iloc[i:chunk_end]

                # 关联处理
                chunk_result = await self._process_point_chunk(point_chunk, cross_section_lines)
                associated_points.extend(chunk_result)

                # 更新进度
                self.progress_monitor.update_progress(chunk_end, len(cross_section_points))

                # 检查是否应该暂停
                if await self._should_pause(context):
                    await self._save_partial_result(context, associated_points, i, len(cross_section_points))
                    raise PipelinePauseException("Pipeline paused by user request")

            # 保存结果
            result_data = {
                'associated_cross_section_points': pd.DataFrame(associated_points),
                'association_statistics': self._calculate_association_statistics(associated_points),
                'performance_metrics': self.performance_tracker.get_metrics()
            }

            context.set_data('point_line_association_result', result_data)
            return result_data

        except Exception as e:
            raise StageExecutionError("PointLineAssociationStage", str(e), context)

        finally:
            self.performance_tracker.end_stage_timer()

    async def _process_point_chunk(self, point_chunk, lines_gdf):
        """处理点数据块"""
        # 异步处理点线关联
        loop = asyncio.get_event_loop()
        return await loop.run_in_executor(
            None,
            self.associator.associate_points_to_lines,
            point_chunk,
            lines_gdf
        )

class LineLineIntersectionStage(BaseProcessingStage):
    """线-线相交处理阶段"""

    async def execute(self, context: PipelineContext):
        """执行线-线相交分析"""
        # 获取横断面和纵断面数据
        h_lines = context.get_data('cross_section_lines')
        v_lines = context.get_data('vertical_section_lines')

        # 初始化相交检测器
        detector = LineIntersectionDetector()
        detector.set_intersection_tolerance(context.config.get('intersection_tolerance', 1e-6))

        # 执行相交检测
        intersections = await self._detect_intersections_with_progress(
            detector, h_lines, v_lines, context
        )

        # 解决多对多相交问题
        resolved_associations = detector.resolve_multiple_intersections(intersections)

        # 保存结果
        result_data = {
            'line_intersections': resolved_associations,
            'intersection_statistics': self._calculate_intersection_statistics(resolved_associations),
            'spatial_relationships': self._build_spatial_relationship_graph(resolved_associations)
        }

        context.set_data('line_line_intersection_result', result_data)
        return result_data

    async def _detect_intersections_with_progress(self, detector, h_lines, v_lines, context):
        """带进度监控的相交检测"""
        intersections = []
        total_v_lines = len(v_lines)

        for i, v_line in v_lines.iterrows():
            # 检测单条纵断面与所有横断面的相交
            line_intersections = detector.find_intersections_for_line(v_line.geometry, h_lines)
            intersections.extend(line_intersections)

            # 更新进度
            progress = (i + 1) / total_v_lines * 100
            context.progress_monitor.update_stage_progress(progress)

            # 允许异步任务调度
            await asyncio.sleep(0.001)

        return intersections
```

```python
class PipelineStateManager:
    """管道状态管理器"""

    def __init__(self, checkpoint_dir='./checkpoints'):
        self.checkpoint_dir = checkpoint_dir
        self.active_sessions = {}
        self._ensure_checkpoint_directory()

    def create_pipeline_session(self, input_config):
        """创建管道会话"""
        session_id = str(uuid.uuid4())
        session_info = {
            'session_id': session_id,
            'input_config': input_config,
            'start_time': datetime.utcnow(),
            'status': 'running',
            'completed_stages': [],
            'current_stage': None,
            'checkpoints': {}
        }

        self.active_sessions[session_id] = session_info
        return session_id

    async def save_checkpoint(self, session_id, stage_index, context):
        """保存检查点"""
        checkpoint_path = os.path.join(self.checkpoint_dir, f"{session_id}_stage_{stage_index}.pkl")

        checkpoint_data = {
            'stage_index': stage_index,
            'context_data': context.serialize(),
            'timestamp': datetime.utcnow(),
            'memory_state': self._capture_memory_state(),
            'performance_metrics': context.get_performance_metrics()
        }

        try:
            with open(checkpoint_path, 'wb') as f:
                pickle.dump(checkpoint_data, f)

            # 更新会话信息
            self.active_sessions[session_id]['checkpoints'][stage_index] = checkpoint_path
            self.active_sessions[session_id]['completed_stages'].append(stage_index)

        except Exception as e:
            raise CheckpointError(f"Failed to save checkpoint for stage {stage_index}: {str(e)}")

    async def load_checkpoint(self, session_id, stage_index=None):
        """加载检查点"""
        if stage_index is None:
            # 加载最新的检查点
            stage_index = max(self.active_sessions[session_id]['completed_stages'])

        checkpoint_path = self.active_sessions[session_id]['checkpoints'][stage_index]

        try:
            with open(checkpoint_path, 'rb') as f:
                checkpoint_data = pickle.load(f)

            # 重建上下文
            context = PipelineContext.deserialize(checkpoint_data['context_data'])
            return context, checkpoint_data

        except Exception as e:
            raise CheckpointError(f"Failed to load checkpoint for stage {stage_index}: {str(e)}")

    async def resume_pipeline(self, session_id):
        """恢复管道执行"""
        session_info = self.active_sessions[session_id]
        last_stage = max(session_info['completed_stages'])

        # 加载最后检查点
        context, checkpoint_data = await self.load_checkpoint(session_id, last_stage)

        # 从下一阶段继续执行
        return context, last_stage + 1
```

```python
class PipelineErrorHandler:
    """管道错误处理器"""

    def __init__(self):
        self.error_strategies = {}
        self.recovery_attempts = {}
        self._initialize_error_strategies()

    def _initialize_error_strategies(self):
        """初始化错误处理策略"""
        self.error_strategies = {
            MemoryError: self._handle_memory_error,
            DataValidationError: self._handle_data_validation_error,
            NetworkError: self._handle_network_error,
            FileSystemError: self._handle_filesystem_error,
            ConcurrencyError: self._handle_concurrency_error
        }

    def handle_stage_error(self, stage_name, error, context):
        """处理阶段错误"""
        error_type = type(error)
        error_info = {
            'stage': stage_name,
            'error_type': error_type.__name__,
            'error_message': str(error),
            'timestamp': datetime.utcnow(),
            'context_snapshot': context.get_snapshot(),
            'recoverable': False,
            'recovery_strategy': None
        }

        # 查找对应的处理策略
        for error_class, strategy in self.error_strategies.items():
            if isinstance(error, error_class):
                error_info = strategy(error, context, error_info)
                break

        # 记录错误
        self._log_error(error_info)

        return error_info

    def _handle_memory_error(self, error, context, error_info):
        """处理内存错误"""
        error_info['recoverable'] = True
        error_info['recovery_strategy'] = 'reduce_chunk_size'

        # 获取当前内存使用情况
        memory_usage = psutil.virtual_memory()
        if memory_usage.percent > 90:
            error_info['suggested_action'] = 'increase_memory_limit'
        else:
            error_info['suggested_action'] = 'reduce_processing_chunk_size'

        return error_info

    async def attempt_recovery(self, error_info, context):
        """尝试错误恢复"""
        recovery_strategy = error_info.get('recovery_strategy')

        if recovery_strategy == 'reduce_chunk_size':
            return await self._recover_by_reducing_chunk_size(context)
        elif recovery_strategy == 'retry_with_exponential_backoff':
            return await self._recover_with_backoff(context)
        elif recovery_strategy == 'switch_to_alternative_algorithm':
            return await self._recover_with_alternative_algorithm(context)
        else:
            raise RecoveryFailedException(f"Unknown recovery strategy: {recovery_strategy}")

    async def _recover_by_reducing_chunk_size(self, context):
        """通过减少块大小恢复"""
        current_chunk_size = context.config.get('chunk_size', 1000)
        new_chunk_size = max(current_chunk_size // 2, 100)

        context.config['chunk_size'] = new_chunk_size
        context.add_recovery_note(f"Reduced chunk size from {current_chunk_size} to {new_chunk_size}")

        return context
```

```python
class ProgressMonitor:
    """实时进度监控器"""

    def __init__(self, update_interval=1.0):
        self.update_interval = update_interval
        self.active_monitors = {}
        self.progress_history = {}

    def start_monitoring(self, pipeline_id):
        """开始监控"""
        monitor_info = {
            'pipeline_id': pipeline_id,
            'start_time': time.time(),
            'stages': {},
            'overall_progress': 0.0,
            'estimated_completion': None,
            'status': 'running'
        }

        self.active_monitors[pipeline_id] = monitor_info
        self.progress_history[pipeline_id] = []

        # 启动监控线程
        threading.Thread(
            target=self._monitoring_loop,
            args=(pipeline_id,),
            daemon=True
        ).start()

    def update_stage_status(self, stage_index, stage_name, status, error_message=None):
        """更新阶段状态"""
        for monitor in self.active_monitors.values():
            if 'stages' in monitor:
                stage_info = monitor['stages'].get(stage_index, {})
                stage_info.update({
                    'name': stage_name,
                    'status': status,
                    'error_message': error_message,
                    'last_update': time.time()
                })
                monitor['stages'][stage_index] = stage_info

    def update_stage_progress(self, pipeline_id, stage_index, progress_percentage):
        """更新阶段进度"""
        if pipeline_id in self.active_monitors:
            monitor = self.active_monitors[pipeline_id]
            if stage_index not in monitor['stages']:
                monitor['stages'][stage_index] = {}

            monitor['stages'][stage_index]['progress'] = progress_percentage
            monitor['stages'][stage_index]['last_update'] = time.time()

            # 更新整体进度
            self._recalculate_overall_progress(pipeline_id)

    def _recalculate_overall_progress(self, pipeline_id):
        """重新计算整体进度"""
        monitor = self.active_monitors[pipeline_id]
        stages = monitor['stages']

        if not stages:
            return

        total_progress = sum(
            stage.get('progress', 0) for stage in stages.values()
        ) / len(stages)

        monitor['overall_progress'] = total_progress

        # 估算完成时间
        if total_progress > 0:
            elapsed_time = time.time() - monitor['start_time']
            estimated_total_time = elapsed_time / (total_progress / 100)
            estimated_completion = monitor['start_time'] + estimated_total_time
            monitor['estimated_completion'] = estimated_completion

    def get_progress_report(self, pipeline_id):
        """获取进度报告"""
        if pipeline_id not in self.active_monitors:
            return None

        monitor = self.active_monitors[pipeline_id]
        current_time = time.time()

        report = {
            'pipeline_id': pipeline_id,
            'overall_progress': monitor['overall_progress'],
            'elapsed_time': current_time - monitor['start_time'],
            'estimated_completion': monitor.get('estimated_completion'),
            'status': monitor['status'],
            'stages': monitor['stages'],
            'performance_metrics': self._calculate_performance_metrics(pipeline_id)
        }

        return report

    def _monitoring_loop(self, pipeline_id):
        """监控循环"""
        while pipeline_id in self.active_monitors:
            try:
                # 更新性能指标
                self._update_performance_metrics(pipeline_id)

                # 记录进度历史
                if pipeline_id in self.progress_history:
                    self.progress_history[pipeline_id].append({
                        'timestamp': time.time(),
                        'progress': self.active_monitors[pipeline_id]['overall_progress']
                    })

                time.sleep(self.update_interval)

            except Exception as e:
                print(f"Monitoring error for pipeline {pipeline_id}: {str(e)}")
                time.sleep(self.update_interval)
```
