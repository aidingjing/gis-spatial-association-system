            start_lat = random.uniform(41, 47)
            end_lon = start_lon + random.uniform(0.01, 0.1)
            end_lat = start_lat + random.uniform(0.01, 0.1)

            lines.append(LineString([(start_lon, start_lat), (end_lon, end_lat)]))
            data.append({
                '名称': f'基准横断面{i}',
                '编号': f'BENCH{i:04d}',
                '河流名': '基准河流',
                '类别': '主河槽'
            })

        return gpd.GeoDataFrame(data, geometry=lines, crs='EPSG:4326')

    def _measure_memory_usage(self):
        """测量当前内存使用"""
        import psutil
        import os
        process = psutil.Process(os.getpid())
        return process.memory_info().rss / 1024**2  # MB

    def _generate_performance_report(self):
        """生成性能测试报告"""
        report = {
            'test_timestamp': time.strftime('%Y-%m-%d %H:%M:%S'),
            'system_info': self._get_system_info(),
            'benchmark_results': self.benchmark_results
        }

        report_path = 'performance_benchmark_report.json'
        import json
        with open(report_path, 'w') as f:
            json.dump(report, f, indent=2)

        print(f"性能基准报告已生成: {report_path}")

    def _analyze_parallel_efficiency(self):
        """分析并行处理效率"""
        single_worker_time = self.benchmark_results.get('workers_1', {}).get('execution_time')
        if not single_worker_time:
            return

        efficiency_report = {}
        for workers in [2, 4, 8]:
            worker_key = f'workers_{workers}'
            if worker_key in self.benchmark_results:
                worker_time = self.benchmark_results[worker_key]['execution_time']
                speedup = single_worker_time / worker_time
                efficiency = speedup / workers * 100

                efficiency_report[worker_key] = {
                    'speedup': speedup,
                    'efficiency_percent': efficiency
                }

        print("并行处理效率分析:")
        for worker_key, metrics in efficiency_report.items():
            print(f"  {worker_key}: 加速比={metrics['speedup']:.2f}, 效率={metrics['efficiency_percent']:.1f}%")
