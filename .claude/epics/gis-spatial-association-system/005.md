class ConfigManager:
    """配置管理系统"""

    def __init__(self):
        self.config = {}
        self.default_config = self._get_default_config()
        self.config_file_path = None

    def _get_default_config(self):
        """获取默认配置"""
        return {
            'processing': {
                'batch_size': 1000,
                'use_parallel': True,
                'max_workers': multiprocessing.cpu_count(),
                'intersection_tolerance': 1e-6,
                'memory_limit_gb': 4,
                'cache_enabled': True
            },
            'coordinate_system': {
                'target_crs': 'auto',
                'auto_detect': True,
                'transformation_method': 'auto'
            },
            'validation': {
                'geometry_validation': True,
                'attribute_validation': True,
                'coordinate_validation': True,
                'strict_mode': False
            },
            'output': {
                'formats': ['shapefile'],
                'generate_report': True,
                'include_metadata': True,
                'compression': False
            },
            'performance': {
                'use_spatial_index': True,
                'chunk_size': 1000,
                'cache_size_mb': 512,
                'parallel_threshold': 1000
            }
        }

    def load_config(self, config_path):
        """加载配置文件"""
        if not os.path.exists(config_path):
            raise FileNotFoundError(f"配置文件不存在: {config_path}")

        try:
            with open(config_path, 'r', encoding='utf-8') as f:
                if config_path.endswith('.yaml') or config_path.endswith('.yml'):
                    loaded_config = yaml.safe_load(f)
                elif config_path.endswith('.json'):
                    import json
                    loaded_config = json.load(f)
                else:
                    raise ValueError(f"不支持的配置文件格式: {config_path}")

            # 合并默认配置和加载的配置
            self.config = self._merge_configs(self.default_config, loaded_config)
            self.config_file_path = config_path

        except Exception as e:
            raise ConfigLoadError(f"加载配置文件失败: {str(e)}")

    def save_config(self, config_path, config=None):
        """保存配置文件"""
        config_to_save = config if config is not None else self.config

        try:
            os.makedirs(os.path.dirname(config_path), exist_ok=True)

            with open(config_path, 'w', encoding='utf-8') as f:
                if config_path.endswith('.yaml') or config_path.endswith('.yml'):
                    yaml.dump(config_to_save, f, default_flow_style=False, allow_unicode=True, indent=2)
                elif config_path.endswith('.json'):
                    import json
                    json.dump(config_to_save, f, indent=2, ensure_ascii=False)
                else:
                    raise ValueError(f"不支持的配置文件格式: {config_path}")

        except Exception as e:
            raise ConfigSaveError(f"保存配置文件失败: {str(e)}")

    def get(self, key_path, default=None):
        """获取配置值"""
        keys = key_path.split('.')
        value = self.config

        try:
            for key in keys:
                value = value[key]
            return value
        except (KeyError, TypeError):
            return default

    def set(self, key_path, value):
        """设置配置值"""
        keys = key_path.split('.')
        config = self.config

        for key in keys[:-1]:
            if key not in config:
                config[key] = {}
            config = config[key]

        config[keys[-1]] = value

    def create_config_template(self, template_path):
        """创建配置模板"""
        template_config = {
            # 处理配置
            'processing': {
                'batch_size': 1000,
                'use_parallel': True,
                'max_workers': 4,
                'intersection_tolerance': 1e-6
            },

            # 输入文件配置
            'input_files': {
                'cross_section_points': 'path/to/cross_section_points.shp',
                'cross_section_lines': 'path/to/cross_section_lines.shp',
                'vertical_section_points': 'path/to/vertical_section_points.shp',
                'vertical_section_lines': 'path/to/vertical_section_lines.shp',
                'prevention_areas': 'path/to/prevention_areas.shp'
            },

            # 输出配置
            'output': {
                'directory': './output',
                'formats': ['shapefile', 'geojson'],
                'generate_report': True
            },

            # 验证配置
            'validation': {
                'geometry_validation': True,
                'attribute_validation': True,
                'coordinate_validation': True
            }
        }

        self.save_config(template_path, template_config)

    def validate_config(self):
        """验证配置有效性"""
        validation_errors = []

        # 检查必需的配置项
        required_paths = [
            'processing.batch_size',
            'input_files.cross_section_points',
            'input_files.cross_section_lines',
            'output.directory'
        ]

        for path in required_paths:
            if self.get(path) is None:
                validation_errors.append(f"缺少必需配置: {path}")

        # 检查配置值的有效性
        batch_size = self.get('processing.batch_size')
        if batch_size is not None and (not isinstance(batch_size, int) or batch_size <= 0):
            validation_errors.append("processing.batch_size 必须是正整数")

        max_workers = self.get('processing.max_workers')
        if max_workers is not None and (not isinstance(max_workers, int) or max_workers <= 0):
            validation_errors.append("processing.max_workers 必须是正整数")

        return validation_errors

    def _merge_configs(self, default, loaded):
        """递归合并配置"""
        result = default.copy()

        for key, value in loaded.items():
            if key in result and isinstance(result[key], dict) and isinstance(value, dict):
                result[key] = self._merge_configs(result[key], value)
            else:
                result[key] = value

        return result
